# 多重for循环时，指定break到具体的循环层

var arr1 = ['a', 'b', 'c']
var arr2= ['1', '2', '3']
outter: for (let i = 0; i < arr1.length; i++) {
  inner: for (let j = 0; j < arr2.length; j++) {
    if (arr2[j] === '2') {
      break outter;
    }
      console.log({ i: arr1[i], j: arr2[j] }); // { i: 'a', j: '1' }
  }
}
